Quick Start

Try It Now

# Universal tool call (MCP-style)
curl -X POST http://defi-mcp.your-server.example.com/api/call \
  -H "Content-Type: application/json" \
  -d '{"tool":"get_token_price","args":{"ids":["bitcoin","ethereum"]}}'

# REST shortcuts
curl http://defi-mcp.your-server.example.com/api/prices?ids=bitcoin,ethereum
curl http://defi-mcp.your-server.example.com/api/search/pepe
curl http://defi-mcp.your-server.example.com/api/gas/all
curl http://defi-mcp.your-server.example.com/api/balance/0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045/multi
curl http://defi-mcp.your-server.example.com/api/top?limit=10

# Python
import requests
r = requests.post("http://defi-mcp.your-server.example.com/api/call",
  json={"tool": "get_multichain_gas", "args": {}})
print(r.json())
